#!/usr/bin/env bash
# Aetheris Integrity Protocol (Final Verification)
set -e

echo "🛡️ Starting Aetheris System Validation..."

# TEST 1: WireGuard Stealth Check
if ! ss -unlp | grep -q ":51820"; then
    echo "❌ FAIL: WireGuard Mesh not listening on UDP 51820"
else
    echo "✅ PASS: Stealth Mesh Active"
fi

# TEST 2: ZFS Encryption & Mount
if zfs get encryption aetheris_vault/secure_data | grep -q "aes-256-gcm"; then
    echo "✅ PASS: Vault is Native Encrypted (AES-256-GCM)"
else
    echo "❌ FAIL: Vault not found or not encrypted"
fi

# TEST 3: Zero-Trust OPA Check (Dry Run)
OPA_RESULT=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:8181/v1/data/aetheris/authz/allow)
if [ "$OPA_RESULT" == "200" ]; then
    echo "✅ PASS: OPA Decision Engine Responsive"
else
    echo "❌ FAIL: OPA Engine Offline (Code: $OPA_RESULT)"
fi

# TEST 4: AI Indexing Latency
echo "Testing Semantic Indexing..."
touch vault/test_file.txt
echo "Aetheris is a sovereign cloud system." > vault/test_file.txt
sleep 2 # Wait for inotify + Rust core + Ollama
if [ -f "data/chroma/chroma.sqlite3" ]; then
    echo "✅ PASS: AI Vector persistence detected"
else
    echo "⚠️ WARNING: Vector DB not yet updated (Index Pending)"
fi
